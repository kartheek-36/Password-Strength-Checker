password = input('ENTER THE PASSWORD: ')
score = 0

islower = 0
isupper = 0
isdigit = 0
for char in password:
    if char.isupper():
        isupper = 1
    elif char.islower():
        islower = 1
    elif char.isdigit():
        isdigit = 1
length = 1 if len(password)>=8  else 0

score = ( length + islower + isupper + isdigit)

if score == 4:
     print('password strenght is strong')

elif score == 3:
     print('password strength is medium')

else:
     print('password strength is low')
